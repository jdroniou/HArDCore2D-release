306  static ReissnerMindlin::SolutionRotationType
307  polynomial_theta = [](const RMParameters & para, const VectorRd & x) -> VectorRd {
308  return VectorRd(
309  pow(x(1),3)*pow(x(1)-1,3)*pow(x(0),2)*pow(x(0)-1,2)*(2*x(0)-1),
310  pow(x(0),3)*pow(x(0)-1,3)*pow(x(1),2)*pow(x(1)-1,2)*(2*x(1)-1)
311  );
312  };
313 
314  static ReissnerMindlin::GradientRotationType
315  polynomial_grad_theta = [](const RMParameters & para, const VectorRd & x) -> Eigen::Matrix2d {
316  Eigen::Matrix2d G = Eigen::Matrix2d::Zero();
317  G(0,0) = 2.*pow(x(0),2)*pow(x(1),3)*pow(x(0)-1.0,2)*pow(x(1)-1.0,3)+2.*x(0)*pow(x(1),3)*(2.*x(0)-1.)*pow(x(0)-1.,2)*pow(x(1)-1.0,3)+pow(x(0),2)*pow(x(1),3)*(2.*x(0)-1.)*(2.*x(0)-2.)*pow(x(1)-1.,3);
318  G(0,1) = 3.*pow(x(0)*x(1),2)*(2.*x(0)-1.)*pow(x(0)-1.,2)*pow(x(1)-1.,3)+3.*pow(x(0),2)*pow(x(1),3)*(2.*x(0)-1.)*pow(x(0)-1.,2)*pow(x(1)-1.,2);
319  G(1,0) = 2.*pow(x(1),2)*pow(x(0),3)*pow(x(1)-1.0,2)*pow(x(0)-1.0,3)+2.*x(1)*pow(x(0),3)*(2.*x(1)-1.)*pow(x(1)-1.,2)*pow(x(0)-1.0,3)+pow(x(1),2)*pow(x(0),3)*(2.*x(1)-1.)*(2.*x(1)-2.)*pow(x(0)-1.,3);
320  G(1,1) = 3.*pow(x(1)*x(0),2)*(2.*x(1)-1.)*pow(x(1)-1.,2)*pow(x(0)-1.,3)+3.*pow(x(1),2)*pow(x(0),3)*(2.*x(1)-1.)*pow(x(1)-1.,2)*pow(x(0)-1.,2);
321 
322  return G;
323  };

325  static ReissnerMindlin::SolutionDisplacementType
326  polynomial_u = [](const RMParameters & para, const VectorRd & x) -> double {
327  double val = 0;
328 
329  val += (1./3.)*pow(x(0),3)*pow(x(0)-1,3)*pow(x(1),3)*pow(x(1)-1,3);
330 
331  val -= ( 2*pow(para.t,2) / (5*(1.-para.nu)) ) *
332  ( pow(x(1),3)*pow(x(1)-1,3)*x(0)*(x(0)-1)*(5*x(0)*x(0)-5*x(0)+1)
333  + pow(x(0),3)*pow(x(0)-1,3)*x(1)*(x(1)-1)*(5*x(1)*x(1)-5*x(1)+1) );
334 
335  return val;
336  };
337 
338  static ReissnerMindlin::ForcingTermType
339  polynomial_f = [](const RMParameters & para, const VectorRd & x) -> double {
340  double val = 0;
341 
342  val += 12*x(1)*(x(1)-1)*(5*x(0)*x(0)-5*x(0)+1) *
343  ( 2*x(1)*x(1)*(x(1)-1)*(x(1)-1) + x(0)*(x(0)-1)*(5*x(1)*x(1)-5*x(1)+1) );
344  val += 12*x(0)*(x(0)-1)*(5*x(1)*x(1)-5*x(1)+1) *
345  ( 2*x(0)*x(0)*(x(0)-1)*(x(0)-1) + x(1)*(x(1)-1)*(5*x(0)*x(0)-5*x(0)+1) );
346 
347  return val * para.E / (12* (1.-para.nu*para.nu) );
348  };

351  // Analytical solution (see paper)
352  // (BC are not zero here)
353 
354  static std::function<double(const VectorRd &)>
355  an_g = [](const VectorRd &x)->double { return sin(PI*x(0))*sin(PI*x(1)); };
356 
357  static std::function<VectorRd(const VectorRd &)>
358  an_GRAD_g = [](const VectorRd & x) -> VectorRd {
359  return VectorRd( PI*cos(PI*x(0))*sin(PI*x(1)), PI*sin(PI*x(0))*cos(PI*x(1)) );
360  };
361 
362  static std::function<MatrixRd(const VectorRd &)>
363  an_HESS_g = [](const VectorRd & x) -> MatrixRd {
364  MatrixRd H = MatrixRd::Zero();
365  H.row(0) << -PI*PI*sin(PI*x(0))*sin(PI*x(1)), PI*PI*cos(PI*x(0))*cos(PI*x(1));
366  H.row(1) << PI*PI*cos(PI*x(0))*cos(PI*x(1)), -PI*PI*sin(PI*x(0))*sin(PI*x(1));
367  return H;
368  };
369 
370  static std::function<double(const VectorRd &)>
371  an_LAPL_g = [](const VectorRd &x)->double { return -2*pow(PI,2)*sin(PI*x(0))*sin(PI*x(1)); };
372 
373  static std::function<double(const VectorRd &)>
374  an_V = [](const VectorRd &x)->double { return x(0) * exp(-x(0))*cos(x(1)); };
375 
376  static std::function<VectorRd(const VectorRd &)>
377  an_GRAD_V = [](const VectorRd & x) -> VectorRd {
378  return VectorRd((1.-x(0))*exp(-x(0))*cos(x(1)), -x(0)*exp(-x(0))*sin(x(1)));
379  };
380 
381  static std::function<MatrixRd(const VectorRd &)>
382  an_HESS_V = [](const VectorRd & x) -> MatrixRd {
383  MatrixRd H = MatrixRd::Zero();
384  H.row(0) << (x(0)-2.)*exp(-x(0))*cos(x(1)), -(1.-x(0))*exp(-x(0))*sin(x(1));
385  H.row(1) << -(1.-x(0))*exp(-x(0))*sin(x(1)), -x(0)*exp(-x(0))*cos(x(1));
386  return H;
387  };
388 
389  static std::function<double(const VectorRd &)>
390  an_LAPL_V = [](const VectorRd &x)->double { return -2.* exp(-x(0))*cos(x(1)); };

393  static ReissnerMindlin::SolutionDisplacementType
394  an_v = [](const RMParameters & para, const VectorRd & x) -> double {
395  return pow(para.t,3) * an_V(x/para.t) + an_g(x);
396  };

398  static ReissnerMindlin::SolutionRotationType
399  an_GRAD_v = [](const RMParameters & para, const VectorRd & x) -> VectorRd {
400  return pow(para.t, 2) * an_GRAD_V(x/para.t) + an_GRAD_g(x);
401  };

404  an_HESS_v = [](const RMParameters & para, const VectorRd & x) -> MatrixRd {
405  return para.t * an_HESS_V(x/para.t) + an_HESS_g(x);
406  };
