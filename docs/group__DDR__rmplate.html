= [](const RMParameters & para, const VectorRd & x) -> double {
double val = 0;
val += 12*x(1)*(x(1)-1)*(5*x(0)*x(0)-5*x(0)+1) *
( 2*x(1)*x(1)*(x(1)-1)*(x(1)-1) + x(0)*(x(0)-1)*(5*x(1)*x(1)-5*x(1)+1) );

}
