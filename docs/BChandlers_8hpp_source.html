69 int offsetIndex(const std::vector<size_t> &c, const int &i){
70  assert( c.size() >= 2 && c.size()%2 == 0);
71 
72  int val = -1;
73  int idx = -1;
74  int offset = 0;
75  // Look for index in c such that c[idx]>i and add offsets (every other pair of indices)
76  do {
77  idx++;
78  if (idx%2 == 1){
79  // c.size() even so no risk of having idx=c.size() here
80  offset += c[idx]-c[idx-1];
81  }
82  }
83  while (idx<int(c.size()) && int(c[idx])<=i);
84 
85  if (idx%2 == 0){
86  val = i - offset;
87  }
88 
89  return val;
90 
91 }

94 Eigen::ArrayXi create_mapDOF(const std::vector<size_t> &c, const size_t N){
95  assert( c.size() >= 2 && c.size()%2 == 0);
96 
97  Eigen::ArrayXi dofs = Eigen::ArrayXi::LinSpaced(N, 0, N-1);
98  Eigen::ArrayXi map = -Eigen::ArrayXi::Ones(N);
99 
100  size_t offset = 0;
101  map.head(c[0]) = dofs.head(c[0]);
102  for (size_t i = 1; i < c.size()-1; i += 2){
103  offset += c[i] - c[i-1];
104  map.segment(c[i], c[i+1]-c[i]) = dofs.segment(c[i], c[i+1]-c[i]) - offset;
105  }
106  offset += c[c.size()-1]-c[c.size()-2];
107  map.tail(N-c[c.size()-1]) = dofs.tail(N-c[c.size()-1]) - offset;
108 
109  return map;
110 
111 }
