663  // Useful formulas to manipulate this basis (all indices starting from 0):
664  // the i-th element in the basis is f_{i%r} E_{i/r}
665  // the matrix E_m has 1 at position (m%N, m/N)
666  // the element f_aE_b is the i-th in the family with i = ( [b/N]N + [b%N] )r + a
667  template<typename ScalarFamilyType, size_t N>
668  class MatrixFamily
669  {
670  public:

680  constexpr static const bool hasAncestor = true;
681  static const bool hasFunction = ScalarFamilyType::hasFunction;
682  static const bool hasGradient = true;
683  static const bool hasDivergence = ( ScalarFamilyType::hasGradient && N==dimspace );
684  static const bool hasCurl = false;
685  static const bool hasHessian = false;

687  MatrixFamily(const ScalarFamilyType & scalar_family)
688  : m_scalar_family(scalar_family),
689  m_E(N*N),
690  m_transposeOperator(Eigen::MatrixXd::Zero(scalar_family.dimension()*N*N, scalar_family.dimension()*N*N))
691  {
692  static_assert(ScalarFamilyType::tensorRank == Scalar,
693  "Vector family can only be constructed from scalar families");
694 
695  // Construct the basis for NxN matrices
696  for (size_t j = 0; j < N; j++){
697  for (size_t i = 0; i < N; i++){
698  m_E[j*N + i] = Eigen::Matrix<double, N, N>::Zero();
699  m_E[j*N + i](i,j) = 1.;
700  }
701  }
702 
703  // Construct the transpose operator
704  for (size_t i = 0; i < dimension(); i++){
705  // The i-th basis element is sent to the j-th basis element s.t. j%r=i%r, (j/r)%N=(i/r)/N, (j/r)/N=(i/r)%N
706  size_t r = m_scalar_family.dimension();
707  size_t j = ( ( (i/r)%N )*N + (i/r)/N ) * r + (i%r);
708  m_transposeOperator(i,j) = 1.;
709  }
710  }

803

805  // To select these rows, we loop over the columns of the underlying matrices, and get the rows in symmetriseOperator()
806  // corresponding to the lower triangular part of each column
807  size_t position = 0;
808  for (size_t i = 0; i<N; i++){
809  // Skip the first i*dim_scalar elements in the column (upper triangle) in symOpe, take the remaining (N-i)*dim_scalar
810  SB.middleRows(position, (N-i)*dim_scalar) = symOpe.middleRows(i*(N+1)*dim_scalar, (N-i)*dim_scalar);
811  // update current position
812  position += (N-i)*dim_scalar;
813  }

816  }

2034  template<typename T>
2035  Eigen::MatrixXd gram_schmidt(
2036  boost::multi_array<T, 2> & basis_eval,
2037  const std::function<double(size_t, size_t)> & inner_product
2038  )
2039  {
2040  auto induced_norm = [inner_product](size_t i)->double
2041  {
2042  return std::sqrt( inner_product(i, i) );
2043  };
2044 
2045  // Number of basis functions
2046  size_t Nb = basis_eval.shape()[0];
2047  // Number of quadrature nodes
2048  size_t Nqn = basis_eval.shape()[1];
2049 
2050  Eigen::MatrixXd B = Eigen::MatrixXd::Zero(Nb, Nb);
2051 
2052  // Normalise the first element
2053  double norm = induced_norm(0);
2054  for (size_t iqn = 0; iqn < Nqn; iqn++) {
2055  basis_eval[0][iqn] /= norm;
2056  } // for iqn
2057  B(0,0) = 1./norm;
2058 
2059  for (size_t ib = 1; ib < Nb; ib++) {
2060  // 'coeffs' represents the coefficients of the ib-th orthogonal function on the ON basis functions 0 to ib-1
2061  Eigen::RowVectorXd coeffs = Eigen::RowVectorXd::Zero(ib);
2062  for (size_t pb = 0; pb < ib; pb++) {
2063  coeffs(pb) = - inner_product(ib, pb);
2064  } // for pb
2065 
2066  // store the values of the orthogonalised ib-th basis function
2067  for (size_t pb = 0; pb < ib; pb++) {
2068  for (size_t iqn = 0; iqn < Nqn; iqn++) {
2069  basis_eval[ib][iqn] += coeffs(pb) * basis_eval[pb][iqn];
2070  } // for iqn
2071  } // for pb
2072 
2073  // normalise ib-th basis function
2074  double norm = induced_norm(ib);
2075  for (size_t iqn = 0; iqn < Nqn; iqn++) {
2076  basis_eval[ib][iqn] /= norm;
2077  } // for iqn
2078  coeffs /= norm;
2079  // Compute ib-th row of B.
2080  // B.topLeftCorner(ib, ib) contains the rows that represent the ON basis functions 0 to ib-1 on
2081  // the original basis. Multiplying on the left by coeffs gives the coefficients of the ON basis
2082  // functions on the original basis functions from 0 to ib-1.
2083  B.block(ib, 0, 1, ib) = coeffs * B.topLeftCorner(ib, ib);
2084  B(ib, ib) = 1./norm;
2085  }
2086  return B;
2087  }
