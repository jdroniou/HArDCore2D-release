303

321

323  return ek * m_scalar_family.function(i % m_scalar_family.dimension(), x);

329  static_assert(hasGradient, "Call to gradient() not available");

333  return G;

341  Eigen::Matrix<double, dimspace, dimspace> rot;
342  rot.row(0) << 0., 1.;

351

353  }

633  compute(

654  template<typename BasisType>
655  static boost::multi_array<typename detail::basis_evaluation_traits<Family<BasisType>, BasisFunction>::ReturnValue, 2>
656  compute(
657  const Family<BasisType> & basis,
658  const QuadratureRule & quad
659  )
660  {
661  typedef detail::basis_evaluation_traits<Family<BasisType>, BasisFunction> traits;
662 
663  boost::multi_array<typename traits::ReturnValue, 2>
664  basis_quad( boost::extents[basis.dimension()][quad.size()] );
665 
666  // Compute values at quadrature note on ancestor basis, and then do a transformation
667  const auto & ancestor_basis = basis.ancestor();
668  boost::multi_array<typename traits::ReturnValue, 2>
669  ancestor_basis_quad = evaluate_quad<BasisFunction>::compute(ancestor_basis, quad);
670 
671  Eigen::MatrixXd M = basis.matrix();
672  for (size_t i = 0; i < size_t(M.rows()); i++) {
673  for (size_t iqn = 0; iqn < quad.size(); iqn++) {
674  basis_quad[i][iqn] = M(i,0) * ancestor_basis_quad[0][iqn];
675  }
676  for (size_t j = 1; j < size_t(M.cols()); j++) {
677  for (size_t iqn = 0; iqn < quad.size(); iqn++) {
678  basis_quad[i][iqn] += M(i,j) * ancestor_basis_quad[j][iqn];
679  }
680  } // for j
681  } // for i
682 
683  return basis_quad;
684  }

697  template<typename T>
698  Eigen::MatrixXd gram_schmidt(
699  boost::multi_array<T, 2> & basis_eval,
700  const std::function<double(size_t, size_t)> & inner_product
701  )
702  {
703  auto induced_norm = [inner_product](size_t i)->double
704  {
705  return std::sqrt( inner_product(i, i) );
706  };
707 
708  // Number of basis functions
709  size_t Nb = basis_eval.shape()[0];
710  // Number of quadrature nodes
711  size_t Nqn = basis_eval.shape()[1];
712 
713  Eigen::MatrixXd B = Eigen::MatrixXd::Zero(Nb, Nb);
714 
715  // Normalise the first element
716  double norm = induced_norm(0);
717  for (size_t iqn = 0; iqn < Nqn; iqn++) {
718  basis_eval[0][iqn] /= norm;
719  } // for iqn
720  B(0,0) = 1./norm;
721 
722  for (size_t ib = 1; ib < Nb; ib++) {
723  // 'coeffs' represents the coefficients of the ib-th orthogonal function on the ON basis functions 0 to ib-1
724  Eigen::RowVectorXd coeffs = Eigen::RowVectorXd::Zero(ib);
725  for (size_t pb = 0; pb < ib; pb++) {
726  coeffs(pb) = - inner_product(ib, pb);
727  } // for pb
728 
729  // store the values of the orthogonalised ib-th basis function
730  for (size_t pb = 0; pb < ib; pb++) {
731  for (size_t iqn = 0; iqn < Nqn; iqn++) {
732  basis_eval[ib][iqn] += coeffs(pb) * basis_eval[pb][iqn];
733  } // for iqn
734  } // for pb
735 
736  // normalise ib-th basis function
737  double norm = induced_norm(ib);
738  for (size_t iqn = 0; iqn < Nqn; iqn++) {
739  basis_eval[ib][iqn] /= norm;
740  } // for iqn
741  coeffs /= norm;
742  // Compute ib-th row of B.
743  // B.topLeftCorner(ib, ib) contains the rows that represent the ON basis functions 0 to ib-1 on
744  // the original basis. Multiplying on the left by coeffs gives the coefficients of the ON basis
745  // functions on the original basis functions from 0 to ib-1.
746  B.block(ib, 0, 1, ib) = coeffs * B.topLeftCorner(ib, ib);
747  B(ib, ib) = 1./norm;
748  }
749  return B;
750  }

769  template<typename BasisType>
770  Family<BasisType> l2_orthonormalize(
771  const BasisType & basis,
772  const QuadratureRule & qr,
773  boost::multi_array<typename BasisType::FunctionValue, 2> & basis_quad
774  )
775  {
776  // Check that the basis evaluation and quadrature rule are coherent
777  assert ( basis.dimension() == basis_quad.shape()[0] && qr.size() == basis_quad.shape()[1] );
778 
779  // The inner product between the i-th and j-th basis vectors
780  std::function<double(size_t, size_t)> inner_product
781  = [&basis_quad, &qr](size_t i, size_t j)->double
782  {
783  double r = 0.;
784  for(size_t iqn = 0; iqn < qr.size(); iqn++) {
785  r += qr[iqn].w * scalar_product(basis_quad[i][iqn], basis_quad[j][iqn]);
786  } // for iqn
787  return r;
788  };
789 
790  Eigen::MatrixXd B = gram_schmidt(basis_quad, inner_product);
791 
792  return Family<BasisType>(basis, B);
793  }

802  template<typename FunctionValue>
803  Eigen::MatrixXd compute_gram_matrix(const boost::multi_array<FunctionValue, 2> & B1,
804  const boost::multi_array<FunctionValue, 2> & B2,
805  const QuadratureRule & qr,
806  const size_t nrows,

809  )

812  assert ( qr.size() == B1.shape()[1] && qr.size() == B2.shape()[1] );
813  // Check that we don't ask for more members of family than available
814  assert ( nrows <= B1.shape()[0] && ncols <= B2.shape()[0] );
815 
816  // Re-cast quadrature weights into ArrayXd to make computations faster
817  Eigen::ArrayXd qr_weights = Eigen::ArrayXd::Zero(qr.size());
818  for (size_t iqn = 0; iqn < qr.size(); iqn++){
819  qr_weights(iqn) = qr[iqn].w;
