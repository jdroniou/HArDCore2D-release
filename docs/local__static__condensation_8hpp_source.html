46  // We permute the DOFs that are statically condensed
47  Eigen::MatrixXd AT = m_Perm * lsT.first * m_Perm.transpose();
48  Eigen::VectorXd bT = m_Perm * lsT.second;
49 
50  // Extract 4 blocks of AT, bT for static condensation
51  Eigen::MatrixXd A11 = AT.topLeftCorner(m_dim_sys, m_dim_sys);
52  Eigen::MatrixXd A12 = AT.topRightCorner(m_dim_sys, m_dim_sc);
53  Eigen::MatrixXd A21 = AT.bottomLeftCorner(m_dim_sc, m_dim_sys);
54  Eigen::MatrixXd A22 = AT.bottomRightCorner(m_dim_sc, m_dim_sc);
55  Eigen::VectorXd b1 = bT.head(m_dim_sys);
56  Eigen::VectorXd b2 = bT.tail(m_dim_sc);
57 
58  // Create condensed system (AT_reduced, bT_reduced) and SC recovery operator (RT, cT)
59  Eigen::MatrixXd A22inv_A21 = A22.ldlt().solve(A21);
60  Eigen::VectorXd A22inv_b2 = A22.ldlt().solve(b2);
61  Eigen::MatrixXd AT_sys = A11 - A12 * A22inv_A21;
62  Eigen::VectorXd bT_sys = b1 - A12 * A22inv_b2;
63  Eigen::MatrixXd AT_sc = -A22inv_A21;
64  Eigen::VectorXd bT_sc = A22inv_b2;
65 
66  return std::make_tuple(AT_sys, bT_sys, AT_sc, bT_sc);
