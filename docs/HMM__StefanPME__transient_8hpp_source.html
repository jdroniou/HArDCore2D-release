158 HMM_StefanPME_Transient::HMM_StefanPME_Transient(HybridCore &hmm, tensor_function_type kappa, source_function_type source, BoundaryConditions BC, solution_function_type exact_solution, grad_function_type grad_exact_solution, TestCaseNonLinearity::nonlinearity_function_type zeta, double weight, std::string solver_type, std::ostream & output)
159  : hmm(hmm),
160  kappa(kappa),
161  source(source),
162  m_BC(BC),
163  exact_solution(exact_solution),
164  grad_exact_solution(grad_exact_solution),
165  zeta(zeta),
166  _weight(weight),
167  solver_type(solver_type),
168  m_output(output) {
169 
170  //---- Compute diffusion and mass matrix (do not change during time stepping or Newton) ------//
171  const Mesh* mesh = hmm.get_mesh();
172  DiffT.resize(mesh->n_cells());
173  MassT.resize(mesh->n_cells());
174  // Source is pre-allocated but not computed as it changes with each time iteration
175  RightHandSideT.resize(mesh->n_cells());
176  for (size_t iT = 0; iT < mesh->n_cells(); iT++) {
177  size_t n_edgesT = mesh->cell(iT)->n_edges();
178  size_t n_local_dofs = 1 + n_edgesT;
179  double measT = mesh->cell(iT)->measure();
180  DiffT[iT] = diffusion_operator(iT);
181  MassT[iT] = Eigen::MatrixXd::Zero(n_local_dofs, n_local_dofs);
182  MassT[iT](0,0) = (1-_weight) * measT;
183  MassT[iT].bottomRightCorner(n_edgesT, n_edgesT) = _weight * (measT / n_edgesT) * Eigen::MatrixXd::Identity(n_edgesT, n_edgesT);
184  }
185 }

187 // PERFORM ONE TIME ITERATION
188 UVector HMM_StefanPME_Transient::iterate(const double tps, const double dt, const UVector& Xn) {
189 
190  boost::timer::cpu_timer timer;
191  boost::timer::cpu_timer timerint;
192  const auto mesh = hmm.get_mesh();
193  size_t n_edges_dofs = mesh->n_edges();
194  size_t n_cell_dofs = mesh->n_cells();
195 
196  //----- COMPUTE SOURCE: load + previous time -----//
197  for (size_t iT = 0; iT < mesh->n_cells(); iT++) {
198  RightHandSideT[iT] = dt * load_operator(tps, iT) + MassT[iT]*Xn.restr(iT);
199  }
200 
201  //----- PARAMETERS FOR NEWTON ------------//
202  constexpr double tol = 1e-8;
203  constexpr size_t maxiter = 400;
204  // relaxation parameter
205  double relax = 1;
206  size_t iter = 0;
207  UVector Xhprev = UVector(Eigen::VectorXd::Zero(n_cell_dofs + n_edges_dofs), *hmm.get_mesh(), 0, 0);
208  UVector Xh = UVector(Eigen::VectorXd::Zero(n_cell_dofs + n_edges_dofs), *hmm.get_mesh(), 0, 0);
209 
210  // Vector of Dirichlet BC, either on u (if weight>0) or zeta(u) (if weight=0)
211  Eigen::VectorXd DirBC = Eigen::VectorXd::Zero(n_cell_dofs + n_edges_dofs);
212  for (size_t ibF = 0; ibF < mesh->n_b_edges(); ibF++){
213  Edge* edge = mesh->b_edge(ibF);
214  if (m_BC.type(*edge)=="dir"){
215  size_t iF = edge->global_index();
216  QuadratureRule quadF = generate_quadrature_rule(*edge, 5);
217  for (QuadratureNode quadrule : quadF){
218  if (_weight == 0){
219  DirBC(n_cell_dofs + iF) += quadrule.w * zeta(exact_solution(tps, quadrule.vector()), "fct");
220  }else{
221  DirBC(n_cell_dofs + iF) += quadrule.w * exact_solution(tps, quadrule.vector());
222  }
223  }
224  // Take average
225  DirBC(n_cell_dofs + iF) /= mesh->b_edge(ibF)->measure();
226  }
227  }
228  Xhprev.asVectorXd() = DirBC;
229 
230  // ---- NEWTON ITERATONS ------ //
231  Eigen::VectorXd RES = residual_scheme(dt, Xhprev); // Scheme is F(X)=C, then RES = C - F(u_prev)
232  std::vector<double> residual(maxiter, 1.0);
233  residual[iter] = RES.norm();
234 
235  while ( (iter < maxiter) && (residual[iter] > tol) ){
236  iter++;
237 
238  // System matrix and initialisation of RHS (only on the edge dofs)
239  Eigen::SparseMatrix<double> GlobMat(n_edges_dofs, n_edges_dofs);
240  std::vector<Eigen::Triplet<double>> triplets_GlobMat;
241  Eigen::VectorXd GlobRHS = RES.tail(n_edges_dofs);
242 
243  // Static condensation: matrix and source to recover cell unknowns
244  Eigen::SparseMatrix<double> ScMat(n_cell_dofs, n_edges_dofs);
245  std::vector<Eigen::Triplet<double>> triplets_ScMat;
246  Eigen::VectorXd ScRHS = Eigen::VectorXd::Zero(n_cell_dofs);
247 
248  // Assemble local contributions
249  for (size_t iT = 0; iT < mesh->n_cells(); iT++) {
250  Cell* cell = mesh->cell(iT);
251  size_t n_edgesT = cell->n_edges();
252 
253  // Local static condensation of element unknowns
254  Eigen::VectorXd XTprev = Xhprev.restr(iT);
255  Eigen::VectorXd ZetaprimeXTprev = apply_nonlinearity(XTprev, "der");
256  Eigen::MatrixXd MatZetaprimeXTprev = Eigen::MatrixXd::Zero(1+n_edgesT, 1+n_edgesT);
257 
258  for (size_t i = 0; i < 1 + n_edgesT; i++){
259  MatZetaprimeXTprev(i, i) = ZetaprimeXTprev(i);
260  }
261 
263  Eigen::MatrixXd MatT = dt * DiffT[iT] * MatZetaprimeXTprev + MassT[iT];
264  Eigen::MatrixXd ATT = MatT.topLeftCorner(1, 1);
265  Eigen::MatrixXd ATF = MatT.topRightCorner(1, n_edgesT);
266  Eigen::MatrixXd AFT = MatT.bottomLeftCorner(n_edgesT, 1);
267  Eigen::MatrixXd AFF = MatT.bottomRightCorner(n_edgesT, n_edgesT);
268 
269  Eigen::PartialPivLU<Eigen::MatrixXd> invATT;
270  invATT.compute(ATT);
271 
272  Eigen::MatrixXd invATT_ATF = invATT.solve(ATF);
273  Eigen::VectorXd RES_cell = RES.segment(iT, 1);
274  Eigen::VectorXd invATT_RES_cell = invATT.solve(RES_cell);
275 
276  // Local matrix and right-hand side on the face unknowns
277  Eigen::MatrixXd MatF = Eigen::MatrixXd::Zero(n_edgesT, n_edgesT);
278  Eigen::VectorXd bF = Eigen::VectorXd::Zero(n_edgesT);
279  MatF = AFF - AFT * invATT_ATF;
280  bF = - AFT * invATT_RES_cell; // only additional RHS coming from static condensation, beyond RES
281 
282  // Assemble static condensation operator
283  ScRHS.segment(iT, 1) = invATT_RES_cell;
284 
286  for (size_t i = 0; i < 1; i++){
287  size_t iGlobal = iT + i;
288  for (size_t j = 0; j < n_edgesT; j++){
289  size_t jGlobal = cell->edge(j)->global_index();
290  triplets_ScMat.emplace_back(iGlobal, jGlobal, invATT_ATF(i, j));
291  }
292  }
293 
294  // GLOBAL MATRIX and RHS on the edge unknowns, using the matrices MatF obtained after static condensation
295  for (size_t i = 0; i < n_edgesT; i++){
296  size_t iGlobal = cell->edge(i)->global_index();
297  for (size_t j = 0; j < n_edgesT; j++){
298  size_t jGlobal = cell->edge(j)->global_index();
299  triplets_GlobMat.emplace_back(iGlobal, jGlobal, MatF(i, j));
300  }
301  GlobRHS(iGlobal) += bF(i);
302  }
303  }
304 
305 
306  // Assemble the global linear system (without BC), and matrix to recover statically-condensed cell dofs
307  GlobMat.setFromTriplets(std::begin(triplets_GlobMat), std::end(triplets_GlobMat));
308  ScMat.setFromTriplets(std::begin(triplets_ScMat), std::end(triplets_ScMat));
309 
310  // Dirichlet boundary conditions are trivial. Newton requires to solve
311  // F'(X^k) (X^{k+1} - X^k) = B - F(X^k)
312  // Both X^k and X^{k+1} have fixed Dirichlet values on boundary edges, so the system we solve
313  // is Dirichlet homogeneous, and we just select the non-dirichlet edges (first ones in the list)
314  size_t n_edge_unknowns = mesh->n_edges() - m_BC.n_dir_edges();
315  Eigen::SparseMatrix<double> A = GlobMat.topLeftCorner(n_edge_unknowns, n_edge_unknowns);
316 //std::cout << "Number non-zero terms in matrix: " << A.nonZeros() << "\n";
317  Eigen::VectorXd B = GlobRHS.head(n_edge_unknowns);
318 
319  // Solve condensed system and recover cell unknowns. dX = X^{k+1}-X^k//
320  Eigen::BiCGSTAB<Eigen::SparseMatrix<double> > solver;
321  solver.compute(A);
322  Eigen::VectorXd dX_edge_unknowns = solver.solve(B);
323 
324  Eigen::VectorXd dX = Eigen::VectorXd::Zero(n_cell_dofs + n_edges_dofs);
325  dX.segment(n_cell_dofs, n_edge_unknowns) = dX_edge_unknowns;
326  dX.head(n_cell_dofs) = ScRHS - ScMat * dX.tail(n_edges_dofs);
327 
328  // Recover the fixed boundary values and cell unknowns (from static condensation and Newton).
329  // We start by putting the Dirichlet BC at the end.
330  Xh.asVectorXd() = DirBC;
331  Xh.asVectorXd().head(n_cell_dofs + n_edge_unknowns) = relax*dX.head(n_cell_dofs + n_edge_unknowns) + Xhprev.asVectorXd().head(n_cell_dofs + n_edge_unknowns);
332 
333  // Compute new residual. We might not accept new Xh but start over reducing relax. Otherwise, accept Xh and increase relax
334  // to a max of 1
335  Eigen::VectorXd RES_temp = residual_scheme(dt, Xh);
336  double residual_temp = RES_temp.norm();
337 
338  if (iter > 1 && residual_temp > 10*residual[iter-1]){
339  // We don't update the residual and Xh, but we will do another iteration with relax reduced
340  relax = relax/2.0;
341  iter--;
342  } else {
343  // Increase relax
344  relax = std::min(1.0, 1.2*relax);
345  // Store Xh in Xhprev, compute new residual
346  Xhprev.asVectorXd() = Xh.asVectorXd();
347  RES = RES_temp;
348  residual[iter] = residual_temp;
349  }
350 
351  m_output << " ...Iteration: " << iter << "; residual = " << residual[iter] << "; relax = "<< relax << "\n";
352 
353  } //-------- END NEWTON ITERATIONS -----------//
354  m_nb_newton = iter;
355 
356  return Xh;
357 }

363 Eigen::VectorXd HMM_StefanPME_Transient::apply_nonlinearity(const Eigen::VectorXd& Y, const std::string type) const {
364  // Type="fct", "nlin" or "der" as per the nonlinear function zeta
365  // The function applied to each coefficient of Y depends on its nature. If it's a coefficient appearing in
366  // the mass-lumping (which depends on _weight), the unknown represents u and we apply the full nonlinear function zeta.
367  // Otherwise, the unknown is zeta(u) itself and the function we apply is the identity s->s (with nlin/der: s-> 1)
368  //
369 
370  Eigen::VectorXd ZetaY;
371  // Initialisation depends on "type". We initialise as if the nonlinearity was the identity, we'll change the
372  // appropriate coefficients afterwards
373  if (type == "fct"){
374  ZetaY = Y;
375  }else if (type == "der" || type == "nlin"){
376  ZetaY = Eigen::VectorXd::Ones(Y.size());
377  }else{
378  m_output << "type unknown in apply_nonlinearity: " << type << "\n";
379  exit(EXIT_FAILURE);
380  }
381 
382  const Mesh* mesh = hmm.get_mesh();
383  // We check if Y is a local vector or a global one, this determines the number of cell and edge unknowns
384  size_t n_cell_dofs = 0;
385  if (size_t(Y.size()) == mesh->n_cells() + mesh->n_edges()){
386  n_cell_dofs = mesh->n_cells();
387  }else{
388  n_cell_dofs = 1;
389  }
390  size_t n_edges_dofs = Y.size() - n_cell_dofs;
391 
392  // The type of nonlinearity we apply on each coefficient depends on _weight
393  if (_weight == 0){
394  for (size_t i = 0; i < n_cell_dofs; i++){
395  ZetaY(i) = zeta(Y(i), type);
396  }
397  }else if (_weight == 1){
398  for (size_t i = n_cell_dofs; i < n_cell_dofs + n_edges_dofs; i++){
399  ZetaY(i) = zeta(Y(i), type);
400  }
401  }else{
402  for (size_t i = 0; i < n_cell_dofs + n_edges_dofs; i++){
403  ZetaY(i) = zeta(Y(i), type);
404  }
405  }
406 
407  return ZetaY;
408 }

423 Eigen::MatrixXd HMM_StefanPME_Transient::diffusion_operator(const size_t iT) const {
424 
425  const auto mesh = hmm.get_mesh();
426  size_t dim = mesh->dim();
427  Cell* T = mesh->cell(iT);
428  double mT = T->measure();
429  const size_t n_edgesT = T->n_edges();
430 
431  size_t local_dofs = 1 + n_edgesT;
432  Eigen::MatrixXd StiffT = Eigen::MatrixXd::Zero(local_dofs, local_dofs);
433 
434  // Diffusion in the cell.
435  Eigen::Vector2d xT = T->center_mass();
436  Eigen::Matrix2d kappaT = kappa(xT.x(), xT.y(), T);
437 
438  // Matrix of \nabla_K
439  Eigen::MatrixXd nablaK = Eigen::MatrixXd::Zero(dim, local_dofs);
440  for (size_t iE=0; iE < n_edgesT; iE++){
441  Edge* E = T->edge(iE);
442  double mE = E->measure();
443  nablaK.block(0, iE+1, dim, 1) = mE * T->edge_normal(iE);
444  }
445  nablaK /= mT;
446 
447  // Contribution per diamond
448  for (size_t iE=0; iE < n_edgesT; iE++){
449  Edge* E = T->edge(iE);
450  Eigen::Vector2d xE = E->center_mass();
451  Eigen::Vector2d nTE = T->edge_normal(iE);
452  double mE = E->measure();
453  // Distance center to edge and measure of diamond
454  double dTE = (xE-xT).dot(nTE);
455  double mTE = mE * dTE / dim;
456 
457  // stabilisation (without scaling or normal: u_E-u_K-nablaK u . (xE-xK))
458  Eigen::MatrixXd stabE = Eigen::MatrixXd::Zero(1, local_dofs);
459  stabE(0) = -1.0;
460  stabE(1+iE) = 1.0;
461  stabE -= (xE-xT).transpose() * nablaK;
462 
463  // Complete gradient with stabilisation
464  Eigen::MatrixXd GRAD = nablaK + (pow(dim, 0.5)/dTE) * nTE * stabE;
465 
466  // Contribution to stiffness
467  StiffT += mTE * GRAD.transpose() * kappaT * GRAD;
468  }
469 
470  return StiffT;
471 }

478 Eigen::VectorXd HMM_StefanPME_Transient::load_operator(const double tps, const size_t iT) const {
479 
480  const auto mesh = hmm.get_mesh();
481  Cell* T = mesh->cell(iT);
482  size_t n_edgesT = T->n_edges();
483  size_t local_dofs = 1 + n_edgesT;
484 
485  // Beware, MassT[iT] must have been created
486  if (MassT.size() < iT || size_t(MassT[iT].rows()) != local_dofs){
487  m_output << "Called load_operator without creating MassT[iT]\n";
488  exit(EXIT_FAILURE);
489  }
490  Eigen::VectorXd fvec = Eigen::VectorXd::Zero(1 + n_edgesT);
491  for (size_t i = 0; i < local_dofs; i++){
492  VectorRd node;
493  if (i == 0){
494  node = T->center_mass();
495  }else{
496  node = T->edge(i-1)->center_mass();
497  }
498  fvec(i) = source(tps, node, T);
499  }
500  // Contribution of source to loading term
501  Eigen::VectorXd load = MassT[iT] * fvec;
502 
503  // Adding Neumann boundary conditions
504  // The boundary conditions coming from Neumann edges are computed based on the piecewise constant
505  // discrete trace: if g is the value of the outer flux,
506  // \int g v -> \int g Tv
507  // where (Tv)_e = average of v on e, for e Neumann edge.
508  // Only edge-based basis functions therefore have a contribution, which will simply be \int_e g.
509  if (T->is_boundary()){
510  for (size_t ilF = 0; ilF < T->n_edges(); ilF++) {
511  Edge* edge = T->edge(ilF);
512  if (m_BC.type(*edge)=="neu"){
513  if (edge->is_boundary()){
514  const auto& nTF = T->edge_normal(ilF);
515  QuadratureRule quadF = generate_quadrature_rule(*edge, 5);
516  std::function<double(VectorRd)> Kgrad_n = [&](VectorRd p){
517  return zeta(exact_solution(tps, p), "der") * nTF.dot(kappa(p.x(),p.y(),T) * grad_exact_solution(tps, p, T));
518  };
519  for (QuadratureNode qr : quadF){
520  load(DimPoly<Cell>(1) + ilF) += qr.w * Kgrad_n(qr.vector());
521  }
522  }
523  }
524  }
525  }
526 
527  return load;
528 }

534 Eigen::VectorXd HMM_StefanPME_Transient::residual_scheme(const double dt, const UVector& Xh) const{
535 
536  const Mesh* mesh = hmm.get_mesh();
537  Eigen::VectorXd RES = Eigen::VectorXd::Zero(mesh->n_cells() + mesh->n_edges());
538 
539  // Compute the residual: C - (dt * DIFF * ZetaXh + MASS * Xh), where DIFF is the diffusion matrix, MASS is the mass matrix and C is the source term
540  size_t n_cell_dofs = mesh->n_cells();
541  for (size_t iT = 0; iT < mesh->n_cells(); iT++){
542  Cell* T = mesh->cell(iT);
543  size_t n_edgesT = T->n_edges();
544  // Local unknown in the cell, and zeta of these unknowns
545  Eigen::VectorXd XT = Xh.restr(iT);
546  Eigen::VectorXd ZetaXT = apply_nonlinearity(XT, "fct");
547 
548  Eigen::VectorXd REST = RightHandSideT[iT] - (dt * DiffT[iT] * ZetaXT + MassT[iT] * XT);
549  RES(iT) += REST(0);
550  for (size_t ilE = 0; ilE < n_edgesT; ilE++){
551  size_t iE = T->edge(ilE)->global_index();
552  RES(n_cell_dofs + iE) += REST(1+ilE);
553  }
554  }
555 
556  // No residual on Dirichlet edges (do not correspond to equations of the system)
557  RES.tail(m_BC.n_dir_edges()) = Eigen::VectorXd::Zero(m_BC.n_dir_edges());
558
