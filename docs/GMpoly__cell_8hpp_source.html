143 template<typename BasisType1, typename BasisType2, size_t N>
144 Eigen::MatrixXd GramMatrix(
145  const Cell& T,
146  const TensorizedVectorFamily<BasisType1, N> & basis1,
147  const TensorizedVectorFamily<BasisType2, N> & basis2,
148  MonomialCellIntegralsType mono_int_map = {}
149  )
150  {
151  Eigen::MatrixXd gm = Eigen::MatrixXd::Zero(basis1.dimension(), basis2.dimension());
152 
153  Eigen::MatrixXd anc_gm = GramMatrix(T, basis1.ancestor(), basis2.ancestor(), mono_int_map);
154  size_t dim1 = anc_gm.rows();
155  size_t dim2 = anc_gm.cols();
156 
157  for (size_t i=0; i<N; i++){
158  gm.block(i*dim1, i*dim2, dim1, dim2) = anc_gm;
159  }
160  return gm;
161  };

255 Eigen::MatrixXd GramMatrix(const Cell& T,
256  const BasisType1 & basis1,
257  const BasisType2 & basis2,
258  MonomialCellIntegralsType mono_int_map = {}
259  )
260  {
261  // If no ancestor is to be used, we shouldn't be in this overload
262  static_assert(useAncestor<BasisType1>() || useAncestor<BasisType2>(), "No method to compute this Gram matrix");
263 
264  if constexpr (!useAncestor<BasisType1>() && useAncestor<BasisType2>()) {
265  return transformGM(basis2, 'C', GramMatrix(T, basis1, basis2.ancestor(), mono_int_map) );

267  return transformGM(basis1, 'R', GramMatrix(T, basis1.ancestor(), basis2, mono_int_map) );
268  } else {
269  return transformGM(basis1, 'R', transformGM(basis2, 'C', GramMatrix(T, basis1.ancestor(), basis2.ancestor(), mono_int_map) ) );
