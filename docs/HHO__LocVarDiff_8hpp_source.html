219  void HHO_LocVarDiff::assemble(HybridCore &hho) {
220 
221  boost::timer::cpu_timer timer; // Time the matrix assembly
222  timer.start();
223  const Mesh* mesh = hho.get_mesh();
224 
225  //--------------- PREPARE SYSTEM ------------------------//
226 
227  // Global triplets for: system matrix, static condensaion/barycentric elimination
228  std::vector<Eigen::Triplet<double>> triplets_GlobMat;
229  std::vector<Eigen::Triplet<double>> triplets_ScBe;
230 
231  // Local bilinear form, triplets and source term (one per cell)
232  aT.resize(mesh->n_cells());
233  std::vector<std::vector<Eigen::Triplet<double>>> cell_triplets_GlobMat;
234  std::vector<std::vector<Eigen::Triplet<double>>> cell_triplets_ScBe;
235  std::vector<Eigen::VectorXd> cell_source(mesh->n_cells());
236  cell_triplets_GlobMat.resize(mesh->n_cells());
237  cell_triplets_ScBe.resize(mesh->n_cells());
238  size_t size_triplets_GlobMat = 0;
239  size_t size_triplets_ScBe = 0;
240 
241  //-------------- ASSEMBLE LOCAL CONTRIBUTIONS -------------//
242 
243  // Function to create local contribution between cell start and cell end-1
244  std::function<void(size_t, size_t)> construct_all_local_contributions
245  = [&](size_t start, size_t end)->void
246  {
247  for (size_t iT = start; iT < end; iT++) {
248  Cell* iCell = mesh->cell(iT);
249 
250  // Total number of edge degrees of freedom local to this cell (adjacent edges to the cell)
251  size_t nlocal_edges = iCell->n_edges();
252  size_t edge_dofs = nlocal_edges * m_nlocal_edge_dofs;
253 
254  // Local bilinear form and source term
255  size_t doeT = std::max( std::max(m_K,m_Ldeg) + m_K+1 , 2*m_K + _deg_kappa );
256  size_t doeF = 2*m_K + 1;
257  ElementQuad elquad(hho, iT, doeT, doeF);
258 
259  aT[iT] = diffusion_operator(hho, iT, elquad);
260  Eigen::VectorXd bT = load_operator(hho, iT, elquad);
261 
262  // Local matrix and right-hand side on the edge unknowns
263  Eigen::MatrixXd MatF = Eigen::MatrixXd::Zero(edge_dofs,edge_dofs);
264  cell_source[iT] = Eigen::VectorXd::Zero(edge_dofs);
265 
266  if (m_L>=0) {
267  // STATIC CONDENSATION OF ELEMENT UNKNOWNS
268 
269  // Perform static condensation
270  Eigen::MatrixXd ATT = aT[iT].topLeftCorner(m_nlocal_cell_dofs, m_nlocal_cell_dofs);
271  Eigen::MatrixXd ATF = aT[iT].topRightCorner(m_nlocal_cell_dofs, edge_dofs);
272  Eigen::MatrixXd AFF = aT[iT].bottomRightCorner(edge_dofs, edge_dofs);
273 
274  Eigen::PartialPivLU<Eigen::MatrixXd> invATT;
275  invATT.compute(ATT);
276 
277  Eigen::MatrixXd invATT_ATF = invATT.solve(ATF);
278  Eigen::VectorXd invATT_bTcell = invATT.solve(bT.head(m_nlocal_cell_dofs));
279  MatF = AFF - ATF.transpose() * invATT_ATF;
280 
281  cell_source[iT] = bT.tail(edge_dofs) - ATF.transpose() * invATT_bTcell;
282 
283  // Assemble local triplets for static condensation operator
284  ScRHS.segment(iT * m_nlocal_cell_dofs, m_nlocal_cell_dofs) = invATT_bTcell;
285  for (size_t i = 0; i < m_nlocal_cell_dofs; i++) {
286  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
287  const size_t jF = iCell->edge(jlF)->global_index();
288  for (size_t jk = 0; jk < m_nlocal_edge_dofs; jk++) {
289  const size_t jLocal = jlF * m_nlocal_edge_dofs + jk;
290  const size_t jGlobal = jF * m_nlocal_edge_dofs + jk;
291  cell_triplets_ScBe[iT].emplace_back(iT * m_nlocal_cell_dofs + i, jGlobal, invATT_ATF(i, jLocal));
292  }
293  }
294  }
295  size_triplets_ScBe += cell_triplets_ScBe[iT].size();
296 
297  } else {
298  // BARYCENTRIC ELIMINATION OF ELEMENT UNKNOWNS
299  // Create reduction matrix: 1+nlocal_edges * nlocal_edges matrix with the coefficients on the first row, and the identity below. When multiplied by the edge unknowns, return cell and edge unknowns
300  // Note that the basis functions are constant, but not necessarily assumed to be one (which is not the case after orthonormalisation for example), which is why we have to adjust the first row.
301  Eigen::MatrixXd red_matT = Eigen::MatrixXd::Zero(1+nlocal_edges,nlocal_edges);
302  red_matT.row(0) = hho.compute_weights(iT);
303  VectorRd xT = iCell->center_mass();
304  double phiT_cst = hho.CellBasis(iT).function(0, xT);
305  for (size_t ilF = 0; ilF < nlocal_edges; ilF++){
306  VectorRd xF = iCell->edge(ilF)->center_mass();
307  size_t iF = iCell->edge(ilF)->global_index();
308  double phiF_cst = hho.EdgeBasis(iF).function(0, xF);
309  red_matT(0,ilF) *= phiF_cst / phiT_cst;
310  }
311  red_matT.bottomRightCorner(nlocal_edges,nlocal_edges) = Eigen::MatrixXd::Identity(nlocal_edges,nlocal_edges);
312 
313  cell_source[iT] = red_matT.transpose() * bT;
314  MatF = red_matT.transpose() * aT[iT] * red_matT;
315 
316  // Assemble local triplets for barycentric combination to recover cell unknown
317  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
318  const size_t jF = iCell->edge(jlF)->global_index();
319  const size_t jGlobal = jF * m_nlocal_edge_dofs;
320  cell_triplets_ScBe[iT].emplace_back(iT, jGlobal, red_matT(0,jlF));
321  }
322  size_triplets_ScBe += cell_triplets_ScBe[iT].size();
323 
324  }
325 
326  // Assemble local triplets for scheme's matrix and source term
327  for (size_t ilF = 0; ilF < nlocal_edges; ilF++) {
328  const size_t iF = iCell->edge(ilF)->global_index();
329  for (size_t ik = 0; ik < m_nlocal_edge_dofs; ik++) {
330  const size_t iLocal = ilF * m_nlocal_edge_dofs + ik;
331  const size_t iGlobal = iF * m_nlocal_edge_dofs + ik;
332  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
333  const size_t jF = iCell->edge(jlF)->global_index();
334  for (size_t jk = 0; jk < m_nlocal_edge_dofs; jk++) {
335  const size_t jLocal = jlF * m_nlocal_edge_dofs + jk;
336  const size_t jGlobal = jF * m_nlocal_edge_dofs + jk;
337  cell_triplets_GlobMat[iT].emplace_back(iGlobal, jGlobal, MatF(iLocal, jLocal));
338  }
339  }
340  }
341  }
342  size_triplets_GlobMat += cell_triplets_GlobMat[iT].size();
343 
344  }
345  }; // End function to construct local contributions
346 
347  // Running the local constructions in parallel
348  parallel_for(mesh->n_cells(), construct_all_local_contributions, m_use_threads);
349 
350  // Assemble local contribution into global matrix
351  triplets_ScBe.reserve(size_triplets_ScBe);
352  triplets_GlobMat.reserve(size_triplets_GlobMat);
353  for (size_t iT = 0; iT < mesh->n_cells(); iT++){
354  for (size_t i = 0; i < cell_triplets_ScBe[iT].size(); i++){
355  triplets_ScBe.push_back(cell_triplets_ScBe[iT][i]);
356  }
357  for (size_t i = 0; i < cell_triplets_GlobMat[iT].size(); i++){
358  triplets_GlobMat.push_back(cell_triplets_GlobMat[iT][i]);
359  }
360  Cell& T = *mesh->cell(iT);
361  for (size_t ilF = 0; ilF < T.n_edges(); ilF++) {
362  const size_t iF = T.edge(ilF)->global_index();
363  for (size_t ik = 0; ik < m_nlocal_edge_dofs; ik++) {
364  const size_t iLocal = ilF * m_nlocal_edge_dofs + ik;
365  const size_t iGlobal = iF * m_nlocal_edge_dofs + ik;
366  GlobRHS(iGlobal) += cell_source[iT](iLocal);
367  }
368  }
369  }
370 
371  if (m_BC.name()=="Neumann"){
372  // Neumann BC: remove a row in the matrix and fix the first degree of freedom
373  triplets_GlobMat.erase(std::remove_if(std::begin(triplets_GlobMat), std::end(triplets_GlobMat),
374  [](const auto& x) { return (x.row() == 0); }), std::end(triplets_GlobMat));
375  triplets_GlobMat.emplace_back(0, 0, 1);
376  GlobRHS(0) = 0;
377  }
378 
379  // Assemble the global linear system (without BC), and matrix to recover statically-condensed cell dofs
380  GlobMat.setFromTriplets(std::begin(triplets_GlobMat), std::end(triplets_GlobMat));
381  ScBeMat.setFromTriplets(std::begin(triplets_ScBe), std::end(triplets_ScBe));
382 
383  // Record assembly time
384  // _assembly_time = timer.elapsed().user + timer.elapsed().system;
385  _assembly_time = timer.elapsed().wall;
386 
387  }

389  UVector HHO_LocVarDiff::solve(HybridCore& hho) {
390  const Mesh* mesh = hho.get_mesh();
391  boost::timer::cpu_timer timer; // Time the matrix assembly
392  timer.start();
393 
394  //-------------- TREATMENT OF BOUNDARY CONDITIONS -------------//
395 
396  // If Dirichlet, the final system is only posed on the interior edge unknowns and we have to subtract from the source
397  // term the contribution of the boundary values
398  // If Neumann, the final system is posed on all edge unknowns
399 
400  size_t n_unknowns = 0;
401  size_t n_fixed_dofs = 0;
402  Eigen::VectorXd B;
403  Eigen::VectorXd UDir;
404 
405  if (m_BC.name() != "Neumann"){
406  // Dirichlet boundary conditions
407  n_unknowns = m_nnondir_edge_dofs;
408  n_fixed_dofs = m_ndir_edge_dofs;
409  SysMat = GlobMat.topLeftCorner(n_unknowns, n_unknowns);
410 
411  // Boundary value: UDir corresponds to the L2 projection of the exact solution on the polynomial spaces on the Dirichlet edges (last BC.n_dir_edges() edges)
412  UDir = Eigen::VectorXd::Zero(n_fixed_dofs);
413  size_t n_dir_edges = m_BC.n_dir_edges();
414  size_t n_nondir_edges = mesh->n_edges() - n_dir_edges;
415  for (size_t idF = 0; idF < n_dir_edges; idF++){
416  Edge* edge = mesh->edge(n_nondir_edges + idF);
417  size_t iF = edge->global_index();
418  QuadratureRule quadF = generate_quadrature_rule(*edge, 2*m_K+2);
419  boost::multi_array<double, 2> phiF_quadF = evaluate_quad<Function>::compute(hho.EdgeBasis(iF), quadF);
420  UDir.segment(idF * m_nlocal_edge_dofs, m_nlocal_edge_dofs) = l2_projection<HybridCore::PolyEdgeBasisType>(exact_solution, hho.EdgeBasis(iF), quadF, phiF_quadF);
421  }
422 
423  B = GlobRHS.segment(0, n_unknowns) - GlobMat.topRightCorner(n_unknowns, n_fixed_dofs) * UDir;
424 
425  } else {
426  // We will solve the complete system
427  n_unknowns = m_ntotal_edge_dofs;
428  SysMat = GlobMat;
429  B = GlobRHS;
430  UDir = Eigen::VectorXd::Zero(n_fixed_dofs);
431  }
432 
433  //-------------- SOLVE CONDENSED SYSTEM -------------//
434 
435  Eigen::VectorXd xF = Eigen::VectorXd::Zero(n_unknowns);
436 
437  // if (solver_type == "ma41") {
438  // Eigen::MA41<Eigen::SparseMatrix<double>, Eigen::VectorXd> solver;
439  // solver.analyzePattern(SysMat);
440  // solver.factorize(SysMat);
441  // xF = solver.solve(B);
442  // } else {
443  Eigen::BiCGSTAB<Eigen::SparseMatrix<double> > solver;
444  solver.compute(SysMat);
445  xF = solver.solve(B);
446  std::cout << " [solver] #iterations: " << solver.iterations() << ", estimated error: " << solver.error() << std::endl;
447  // }
448  _solving_error = (SysMat * xF - B).norm();
449  // Recover the fixed boundary values, cell unknowns (from static condensation/barycentric elimination)
450  Eigen::VectorXd Xh = Eigen::VectorXd::Zero(m_ntotal_dofs);
451  Xh.tail(n_fixed_dofs) = UDir;
452  Xh.segment(m_ntotal_cell_dofs, n_unknowns) = xF;
453  if (m_L>=0) {
454  Xh.head(m_ntotal_cell_dofs) = ScRHS - ScBeMat * Xh.tail(m_ntotal_edge_dofs);
455  } else {
456  Xh.head(m_ntotal_cell_dofs) = ScBeMat * Xh.tail(m_ntotal_edge_dofs);
457  }
458 
459  // Only Neumann: translate to get the proper average
460  if (m_BC.name()=="Neumann"){
461  // Compute average to translate
462  double average = 0.;
463  double total_measure = 0;
464  for (size_t iT = 0; iT < mesh->n_cells(); iT++){
465  Cell& T = *mesh->cell(iT);
466  total_measure += T.measure();
467  HybridCore::PolyCellBasisType basisT = hho.CellBasis(iT);
468  QuadratureRule quadT = generate_quadrature_rule(T, 2*m_K);
469  boost::multi_array<double, 2> phiT_quadT = evaluate_quad<Function>::compute(basisT, quadT);
470  for (size_t i = 0; i < basisT.dimension(); i++){
471  for (size_t iqn = 0; iqn < quadT.size(); iqn++){
472  average += quadT[iqn].w * Xh(iT * m_nlocal_cell_dofs + i) * basisT.function(i, quadT[iqn].vector());
473  }
474  }
475  }
476  double average_exact_sol = 0.0;
477  for (auto& T : mesh->get_cells()){
478  QuadratureRule quadT = generate_quadrature_rule(*T, 2 * m_Ldeg + 2);
479  for (QuadratureNode& qT : quadT){
480  average_exact_sol += qT.w * exact_solution(qT.vector());
481  }
482  }
483  average_exact_sol /= total_measure;
484 
485  // Translate the cells and edges
486  // We compute the interpolant of the constant function "average_exact_sol - average"
487  // and we translate Xh by that amount
488  std::function<double(VectorRd)> AveDiff = [&average_exact_sol,&average](VectorRd x)->double
489  { return average_exact_sol - average;};
490 
491  UVector Cst = hho.interpolate(AveDiff, m_L, m_K, 2*m_K+3);
492  Xh += Cst.asVectorXd();
493  }
494 
495  _solving_time = timer.elapsed().user + timer.elapsed().system; // Record the final solving time
496 
497  return UVector(Xh, *hho.get_mesh(), m_L, m_K);
498  }

504  Eigen::MatrixXd HHO_LocVarDiff::diffusion_operator(HybridCore &hho, const size_t iT, const ElementQuad &elquad) const
505  {
506 
507  boost::timer::cpu_timer timeint;
508 
509  const auto mesh = hho.get_mesh();
510  const size_t dimPKcell = DimPoly<Cell>(m_K);
511  const size_t dimPKcell_vec = mesh->dim() * dimPKcell;
512  Cell* cell = mesh->cell(iT);
513  const size_t nedgesT = cell->n_edges();
514 
515  // Total number of degrees of freedom local to this cell (cell and its adjacent edges)
516  size_t local_dofs = m_nlocal_cell_dofs + nedgesT * m_nlocal_edge_dofs;
517 
518  //------------------- Initialisatons: quadratures, mass matrices... --------------------//
519 
520  // Note: representing the gradient reconstruction GT supposes a basis for (P^k)^d. This basis can be
521  // built from the basis (phi_i)_i of P^k, by working component-by-component:
522  // (phi_1 e_1, ..., phi_N e_1, phi_1 e_2, ..., phi_N e_2, ...),
523  // where e_1=[1 0], e_2=[0 1]... is the canonical basis of R^d
524  // Hence, the first dimPKcell functions are on the first component of R^d, the next dimPKcell on the second
525  // component etc.
526  // We do not explicitly build this basis function, but we compute the values of these functions at the
527  // quadrature nodes, using the values computed for the scalar basis functions
528 
529 
530  // QUADRATURES
531  // Cell quadrature nodes, and values of cell basis functions (up to degree K+1) and gradients thereof.
532  QuadratureRule quadT = elquad.get_quadT();
533  boost::multi_array<double, 2> phiT_quadT = elquad.get_phiT_quadT();
534  boost::multi_array<VectorRd, 2> dphiT_quadT = elquad.get_dphiT_quadT();
535 
536  // Vector basis functions (up to degree K) at the quadrature nodes.
537  boost::multi_array<VectorRd, 2> vec_phiT_quadT = elquad.get_vec_phiT_quadT(m_K);
538 
539  // Diffusion tensor at the quadrature nodes
540  std::vector<MatrixRd> kappaT_quadT(quadT.size());
541  std::transform(quadT.begin(), quadT.end(), kappaT_quadT.begin(),
542  [this,&cell](QuadratureNode qr) -> Eigen::MatrixXd { return kappa(qr.vector(), cell); });
543 
544  // MASS MATRICES
545  // Scalar cell mass matrix (phi_i,phi_j)_T up to degree max(L,K) * (K+1), and
546  // Vector cell mass matrix (Phi_i,Phi_j)_T up to degree K*K [this one is block diagonal]
547  size_t maxdimPKL = std::max(m_nlocal_cell_dofs, dimPKcell);
548 
549  _itime[0] += timeint.elapsed().user + timeint.elapsed().system;
550  timeint.start();
551 
552  Eigen::MatrixXd MTT = compute_gram_matrix(phiT_quadT, phiT_quadT, quadT, maxdimPKL, m_nhighorder_dofs, "sym");
553  Eigen::MatrixXd VecMTT = Eigen::MatrixXd::Zero(dimPKcell_vec, dimPKcell_vec);
554  for (size_t r=0; r < mesh->dim(); r++){
555  VecMTT.block(r*dimPKcell, r*dimPKcell, dimPKcell, dimPKcell) = MTT.topLeftCorner(dimPKcell, dimPKcell);
556  }
557 
558  // Edge mass matrices:
559  // MFF[ilF]: edge-edge mass on edge with local number ilF, up to degree K*K
560  // MFT[ilF]: edge-cell mass on edge with local number ilF, up to degree K*(K+1)
561  // MTT_on_F[ilF]: cell-cell mass on edge with local number ilF, up to degree K*L
562  std::vector<Eigen::MatrixXd> MFF(nedgesT, Eigen::MatrixXd::Zero(m_nlocal_edge_dofs, m_nlocal_edge_dofs));
563  std::vector<Eigen::MatrixXd> MFT(nedgesT, Eigen::MatrixXd::Zero(m_nlocal_edge_dofs, m_nhighorder_dofs));
564  std::vector<Eigen::MatrixXd> MTT_on_F(nedgesT, Eigen::MatrixXd::Zero(dimPKcell, m_nlocal_cell_dofs));
565 
566  for (size_t ilF = 0; ilF < nedgesT; ilF++) {
567 
568  // Edge quadrature nodes and values of cell and edge basis functions (and gradients) at these nodes
569  auto quadF = elquad.get_quadF(ilF);
570  boost::multi_array<double, 2> phiT_quadF = elquad.get_phiT_quadF(ilF);
571  boost::multi_array<double, 2> phiF_quadF = elquad.get_phiF_quadF(ilF);
572 
573  // Mass matrices
574  MFF[ilF] = compute_gram_matrix(phiF_quadF, phiF_quadF, quadF, "sym");
575  MFT[ilF] = compute_gram_matrix(phiF_quadF, phiT_quadF, quadF, "nonsym");
576  MTT_on_F[ilF] = compute_gram_matrix(phiT_quadF, phiT_quadF, quadF, dimPKcell, m_nlocal_cell_dofs, "nonsym");
577  }
578 
579  _itime[1] += timeint.elapsed().user + timeint.elapsed().system;
580  timeint.start();
581 
582  // STIFNESS mass-matrices (and the like): (\nabla phi_i,\nabla phi_j)_T up to degree (K+1)*(K+1)
583  // and (\nabla phi_i, Phi_j)_T up to degree (K+1)*K
584  Eigen::MatrixXd StiffT = compute_gram_matrix(dphiT_quadT, dphiT_quadT, quadT, "sym");
585  Eigen::MatrixXd MdphiT_PhiT = compute_gram_matrix(dphiT_quadT, vec_phiT_quadT, quadT, "nonsym");
586 
587  _itime[2] += timeint.elapsed().user + timeint.elapsed().system;
588  timeint.start();
589 
590  //-------------------- Compute GT, matrix of full gradient reconstruction ---------//
591 
592  // Right-hand side, starting with volumetric term (Phi_i, \nabla phi_j)_T
593  Eigen::MatrixXd RHS_GT = Eigen::MatrixXd::Zero(dimPKcell_vec, local_dofs);
594  RHS_GT.topLeftCorner(dimPKcell_vec, m_nlocal_cell_dofs) = (MdphiT_PhiT.topLeftCorner(m_nlocal_cell_dofs, dimPKcell_vec)).transpose();
595 
596  // Boundary terms
597  for (size_t r=0; r < mesh->dim(); r++){
598  for (size_t ilF = 0; ilF < nedgesT; ilF++) {
599  // Offset for edge unknowns
600  const size_t offset_F = m_nlocal_cell_dofs + ilF * m_nlocal_edge_dofs;
601  const auto& nTF = cell->edge_normal(ilF);
602 
603  // Contribution of cell unknowns, and then edge unknowns on F
604  RHS_GT.block(r*dimPKcell, 0, dimPKcell, m_nlocal_cell_dofs) -= nTF(r) * MTT_on_F[ilF];
605  RHS_GT.block(r*dimPKcell, offset_F, dimPKcell, m_nlocal_edge_dofs) +=
606  nTF(r) * (MFT[ilF].topLeftCorner(m_nlocal_edge_dofs, dimPKcell)).transpose();
607  }
608  }
609 
610  // Compute GT
611  Eigen::MatrixXd GT = (VecMTT.ldlt()).solve(RHS_GT);
612 
613  //------------- Consistent contribution (K GT, GT)_T to the local bilinear form ------//
614 
615 
616  // Weighted mass matrix (K Phi_i, Phi_j)_T of the basis (Phi_i)_i of (P^k)^d
617  // We start by creating K Phi_i at the quadrature nodes
618  boost::multi_array<VectorRd, 2> kappa_vec_phiT_quadT;
619  kappa_vec_phiT_quadT.resize( boost::extents[dimPKcell_vec][quadT.size()] );
620  for (size_t i = 0; i < dimPKcell_vec; i++){
621  for (size_t iqn = 0; iqn < quadT.size(); iqn++){
622  kappa_vec_phiT_quadT[i][iqn] = kappaT_quadT[iqn] * vec_phiT_quadT[i][iqn];
623  }
624  }
625  _itime[3] += timeint.elapsed().user + timeint.elapsed().system;
626  timeint.start();
627 
628  Eigen::MatrixXd kappaVecMTT = compute_gram_matrix(kappa_vec_phiT_quadT, vec_phiT_quadT, quadT, vec_phiT_quadT.shape()[0], vec_phiT_quadT.shape()[0], "sym");
629 
630  _itime[4] += timeint.elapsed().user + timeint.elapsed().system;
631  timeint.start();
632 
633  Eigen::MatrixXd ATF = GT.transpose() * kappaVecMTT * GT;
634 
635 
636  //------------- Compute PT, matrix of potential reconstruction, using GT ------//
637 
638  // We write that \nabla pT = projection on \nabla P^{k+1} of GT, and add the closure relation:
639  // (nabla pT v, nabla w)_T + lambda_T(p_T v,1)_T(w,1)_T = (GT v,\nabla w)_T + lambda_T(v_T,1)_T(w,1)_T
640 
641  // Right-hand side, starting with volumetric term
642  Eigen::MatrixXd RHS_PT = MdphiT_PhiT * GT;
643 
644  // Vector LT of (phi_j,1)_T for phi_j up to degree K+1, and LT^t*LT, for the closure relation
645  Eigen::VectorXd LT = (MTT.row(0)).transpose();
646  Eigen::MatrixXd LTtLT = LT * (LT.transpose());
647  double scalT = StiffT.trace() / std::pow(LT.norm(), 2);
648 
649  // Add closure relation and compute PT
650  RHS_PT.topLeftCorner(m_nhighorder_dofs, m_nlocal_cell_dofs) +=
651  scalT * LTtLT.topLeftCorner(m_nhighorder_dofs, m_nlocal_cell_dofs);
652  Eigen::MatrixXd PT = ((StiffT + scalT*LTtLT).ldlt()).solve(RHS_PT);
653 
654 
655  //-------------------- Compute stabilisation term sT ---------//
656 
657  Eigen::MatrixXd STF = Eigen::MatrixXd::Zero(local_dofs, local_dofs);
658 
659  // Cell residual delta_T^l = pi_T^l (rT uT) - u_T
660 
661  Eigen::MatrixXd MTT_LKp1 = MTT.topLeftCorner(m_nlocal_cell_dofs, m_nhighorder_dofs);
662  Eigen::MatrixXd MTT_LL = MTT.topLeftCorner(m_nlocal_cell_dofs, m_nlocal_cell_dofs);
663  Eigen::MatrixXd deltaTL = MTT_LL.ldlt().solve( MTT_LKp1 * PT );
664  deltaTL.topLeftCorner(m_nlocal_cell_dofs, m_nlocal_cell_dofs) -= Eigen::MatrixXd::Identity(m_nlocal_cell_dofs, m_nlocal_cell_dofs);
665 
666  for (size_t ilF = 0; ilF < nedgesT; ilF++) {
667  // Two options for stabilisation: diameter of edge, or ratio measure cell/measure edge
668  double dTF = cell->edge(ilF)->diam();
669  // double dTF = cell->measure() / cell->edge(ilF)->measure();
670 
671  VectorRd xF = cell->edge(ilF)->center_mass();
672 
673  // auto kappa_TF = kappa(xF, cell).trace();
674 
675  const VectorRd &nTF = cell->edge_normal(ilF);
676  const double kappa_TF = (kappa(xF, cell) * nTF).dot(nTF);
677 
678  // Edge residual delta_TF^k = pi_F^k (rT uT) - u_F
679  Eigen::MatrixXd MFFinv = MFF[ilF].inverse();
680  Eigen::MatrixXd deltaTFK = MFFinv * MFT[ilF] * PT;
681  deltaTFK.block(0, m_nlocal_cell_dofs + ilF * m_nlocal_edge_dofs, m_nlocal_edge_dofs, m_nlocal_edge_dofs) -=
682  Eigen::MatrixXd::Identity(m_nlocal_edge_dofs, m_nlocal_edge_dofs);
683 
684  // Stabilisation term: here, we actually project deltaTL on P^k(F) so, for l=k+1, it actually corresponds to the stabilisation used in HDG methods (see Section 5.1.6 of HHO book)
685  Eigen::MatrixXd deltaTFK_minus_deltaTL = deltaTFK - MFFinv * MFT[ilF].topLeftCorner(m_nlocal_edge_dofs, m_nlocal_cell_dofs) * deltaTL;
686 
687  STF += (kappa_TF / dTF) * deltaTFK_minus_deltaTL.transpose() * MFF[ilF] * deltaTFK_minus_deltaTL;
688  }
689 
690  _itime[5] += timeint.elapsed().user + timeint.elapsed().system;
691 
692  // Adjust local bilinear form with stabilisation term
693  ATF += mesh->dim() * STF;
694 
695  return ATF;
696 
697  }

704  Eigen::VectorXd HHO_LocVarDiff::load_operator(HybridCore &hho, const size_t iT, const ElementQuad &elquad) const {
705  // Load for the cell DOFs (first indices) and edge DOFs (last indices)
706  const auto mesh = hho.get_mesh();
707  Cell* cell = mesh->cell(iT);
708  size_t cell_edge_dofs = m_nlocal_cell_dofs + cell->n_edges()*m_nlocal_edge_dofs;
709  Eigen::VectorXd b = Eigen::VectorXd::Zero(cell_edge_dofs);
710 
711  // Quadrature nodes and values of cell basis functions at these nodes
712  auto quadT = elquad.get_quadT();
713  size_t nbq = quadT.size();
714  boost::multi_array<double, 2> phiT_quadT = elquad.get_phiT_quadT();
715 
716  // Value of source times quadrature weights at the quadrature nodes
717  Eigen::ArrayXd weight_source_quad = Eigen::ArrayXd::Zero(nbq);
718  for (size_t iqn = 0; iqn < nbq; iqn++){
719  weight_source_quad(iqn) = quadT[iqn].w * source(quadT[iqn].vector(), cell);
720  }
721 
722  for (size_t i=0; i < m_nlocal_cell_dofs; i++){
723  for (size_t iqn = 0; iqn < quadT.size(); iqn++){
724  b(i) += weight_source_quad[iqn] * phiT_quadT[i][iqn];
725  }
726  }
727 
728  // Boundary values, if we have a boundary cell
729  if (cell->is_boundary()){
730  // Boundary values only on boundary Neumann edges
731  for (size_t ilF = 0; ilF < cell->n_edges(); ilF++) {
732  Edge* F = cell->edge(ilF);
733  if (m_BC.type(*F)=="neu"){
734  const size_t iF = F->global_index();
735  // BC on boundary edges
736  if (F->is_boundary()){
737  // Offset for edge unknowns
738  const size_t offset_F = m_nlocal_cell_dofs + ilF * m_nlocal_edge_dofs;
739  // Normal to the edge and bases function
740  const auto& nTF = cell->edge_normal(ilF);
741  const auto& basisF = hho.EdgeBasis(iF);
742  // for each DOF of the boundary edge
743  for (size_t i = 0; i < m_nlocal_edge_dofs; i++){
744  QuadratureRule quadF = generate_quadrature_rule(*F, 2*m_K+2);
745  std::function<double(VectorRd)> Kgrad_n = [&](VectorRd p){
746  return nTF.dot(kappa(p,cell) * grad_exact_solution(p,cell)) * basisF.function(i,p);
747  };
748  for (QuadratureNode& qF : quadF){
749  b(offset_F + i) += qF.w * Kgrad_n(qF.vector());
750  }
751  }
752  }
753  }
754  }
755  }
756 
757  return b;
758  }
