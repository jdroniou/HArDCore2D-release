281  template<typename ContinuousFunction>
282  UVector HybridCore::interpolate(const ContinuousFunction& f, int cell_deg, size_t edge_deg, size_t doe) const {
283 
284  size_t cell_deg_pos = std::max(cell_deg, 0);
285  assert(cell_deg <= CellDegree() || edge_deg <= EdgeDegree() || cell_deg > -1);
286 
287  // Sizes of local interpolation
288  size_t n_cell_dofs = DimPoly<Cell>(cell_deg_pos);
289  size_t n_total_cell_dofs = m_mesh->n_cells() * n_cell_dofs;
290  size_t n_edge_dofs = DimPoly<Edge>(edge_deg);
291  size_t n_total_edge_dofs = m_mesh->n_edges() * n_edge_dofs;
292  Eigen::VectorXd XTF = Eigen::VectorXd::Zero( n_total_cell_dofs + n_total_edge_dofs );
293 
294  // Edge projections
295  std::function<void(size_t, size_t)> construct_all_edge_projections
296  = [&](size_t start, size_t end)->void
297  {
298  for (size_t iE = start; iE < end; iE++) {
299  Edge E = *m_mesh->edge(iE);
300 
301  // Compute the L2 projection on edge
302  QuadratureRule quadE = generate_quadrature_rule(E, doe);
303  boost::multi_array<double, 2> phiE_quadE = evaluate_quad<Function>::compute(EdgeBasis(iE), quadE);
304  Eigen::VectorXd UF = l2_projection<PolyEdgeBasisType>(f, EdgeBasis(iE), quadE, phiE_quadE);
305 
306  // Fill in the complete vector of DOFs
307  XTF.segment(n_total_cell_dofs + iE * n_edge_dofs, n_edge_dofs) = UF;
308  }
309  };
310  parallel_for(m_mesh->n_edges(), construct_all_edge_projections, m_use_threads);
311 
312  // Cell projections
313  std::function<void(size_t, size_t)> construct_all_cell_projections
314  = [&](size_t start, size_t end)->void
315  {
316  for (size_t iT = start; iT < end; iT++) {
317  Cell* cell = m_mesh->cell(iT);
318 
319  // L2 projection in cell
320  // Mass matrix in cell
321  QuadratureRule quadT = generate_quadrature_rule(*cell, doe);
322  boost::multi_array<double, 2> phiT_quadT = evaluate_quad<Function>::compute(CellBasis(iT), quadT);
323  Eigen::MatrixXd MT = compute_gram_matrix(phiT_quadT, phiT_quadT, quadT, n_cell_dofs, n_cell_dofs, "sym");
324 
325  // Vector of integrals of f against basis functions
326  Eigen::VectorXd bT = Eigen::VectorXd::Zero(n_cell_dofs);
327  for (size_t i = 0; i < n_cell_dofs; i++) {
328  for (size_t iqn = 0; iqn < quadT.size(); iqn++){
329  bT(i) += quadT[iqn].w * phiT_quadT[i][iqn] * f(quadT[iqn].vector());
330  }
331  }
332 
333  // Vector of coefficients (on cell basis functions) of the L2(T) projection of f
334  Eigen::VectorXd UT = MT.ldlt().solve(bT);
335 
336  // Fill in the complete vector of DOFs
337  size_t offset_T = iT * n_cell_dofs;
338  XTF.segment(offset_T, n_cell_dofs) = UT;
339 
340  // Special case of L=-1, we replace the previously computed cell value with the proper average of edge values
341  if ( cell_deg==-1 ){
342  size_t nedges = cell->n_edges();
343  // Weights corresponding to the values in cells/on edges
344  Eigen::VectorXd barycoefT = compute_weights(iT);
345  // We transform these weights so that they apply to the coefficients on the basis functions
346  VectorRd xT = cell->center_mass();
347  double phiT_cst = CellBasis(iT).function(0, xT);
348  for (size_t ilE = 0; ilE < nedges; ilE++){
349  VectorRd xE = cell->edge(ilE)->center_mass();
350  size_t iE = cell->edge(ilE)->global_index();
351  double phiE_cst = EdgeBasis(iE).function(0, xE);
352  barycoefT(ilE) *= phiE_cst / phiT_cst;
353  }
354 
355  XTF(iT) = 0;
356  for (size_t ilE = 0; ilE < nedges; ilE++) {
357  size_t iE = cell->edge(ilE)->global_index();
358  XTF(iT) += barycoefT(ilE) * XTF(n_total_cell_dofs + iE);
359  }
360  }
361 
362  }
363  };
364  parallel_for(m_mesh->n_cells(), construct_all_cell_projections, m_use_threads);
365 
366  return UVector(XTF, *get_mesh(), cell_deg, edge_deg);
367  }
